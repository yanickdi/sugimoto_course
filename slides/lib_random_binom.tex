    \begin{itemize}
  	\item Erzeugt binomialverteilte Zufallszahl
  	\item $ z := |\left\{ i|u_i<p \right\}|$
  \end{itemize}
  \begin{lstlisting}[language=python]
def random_binom(n, p):
    z = 0
    # create n random numbers
    # between [0,1]:
    for i in range(n):
        rand = random.random()
        z += 1 if rand < p else 0
    return z
\end{lstlisting}
